## Quick start

```ts
import {createApp} from 'vue';
import {VueSpectrumPlugin} from '@vue-spectrum/components';
import '@vue-spectrum/components/styles.css';
import App from './App.vue';

createApp(App).use(VueSpectrumPlugin).mount('#app');
```
